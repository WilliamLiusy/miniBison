START -> COM
COM -> BODY_COM
BODY_COM -> COM ; SEN
BODY_COM -> SEN
SEN -> ID := INT_EXPR
SEN -> IF BOOL_EXPR THEN BODY_COM ELSE BODY_COM
SEN -> WHILE BOOL_EXPR DO BODY_COM
INT_EXPR -> INT_EXPR + INT_EXPR_NXT
INT_EXPR -> INT_EXPR_NXT
INT_EXPR_NXT -> ( INT_EXPR )
INT_EXPR_NXT -> ID
INT_EXPR_NXT -> CONST
BOOL_EXPR -> INT_EXPR < INT_EXPR
